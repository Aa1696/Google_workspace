143   * Returns an immutable map instance containing the given entries. Internally, the returned map
144   * will be backed by an {@link EnumMap}.
145   *
146   * <p>The iteration order of the returned map follows the enum's iteration order, not the order in
147   * which the elements appear in the given map.
148   *
149   * @param map the map to make an immutable copy of
150   * @return an immutable map containing those entries
151   * @since 14.0
152   */
153  @GwtCompatible(serializable = true)
154  public static <K extends Enum<K>, V> ImmutableMap<K, V> immutableEnumMap(
155      Map<K, ? extends V> map) {
156    if (map instanceof ImmutableEnumMap) {
157      @SuppressWarnings("unchecked") // safe covariant cast
158      ImmutableEnumMap<K, V> result = (ImmutableEnumMap<K, V>) map;
159      return result;
160    }
161    Iterator<? extends Entry<K, ? extends V>> entryItr = map.entrySet().iterator();
162    if (!entryItr.hasNext()) {
163      return ImmutableMap.of();
164    }
165    Entry<K, ? extends V> entry1 = entryItr.next();
166    K key1 = entry1.getKey();
167    V value1 = entry1.getValue();
168    checkEntryNotNull(key1, value1);
169    // Do something that works for j2cl, where we can't call getDeclaredClass():
170    EnumMap<K, V> enumMap = new EnumMap<>(singletonMap(key1, value1));
171    while (entryItr.hasNext()) {
172      Entry<K, ? extends V> entry = entryItr.next();
173      K key = entry.getKey();
174      V value = entry.getValue();
175      checkEntryNotNull(key, value);
176      enumMap.put(key, value);
177    }
178    return ImmutableEnumMap.asImmutable(enumMap);
179  }

278  /**
279   * Returns a capacity that is sufficient to keep the map from being resized as long as it grows no
280   * larger than expectedSize and the load factor is ≥ its default (0.75).
281   */
282  static int capacity(int expectedSize) {
283    if (expectedSize < 3) {
284      checkNonnegative(expectedSize, "expectedSize");
285      return expectedSize + 1;
286    }
287    if (expectedSize < Ints.MAX_POWER_OF_TWO) {
288      // This seems to be consistent across JDKs. The capacity argument to HashMap and LinkedHashMap
289      // ends up being used to compute a "threshold" size, beyond which the internal table
290      // will be resized. That threshold is ceilingPowerOfTwo(capacity*loadFactor), where
291      // loadFactor is 0.75 by default. So with the calculation here we ensure that the
292      // threshold is equal to ceilingPowerOfTwo(expectedSize). There is a separate code
293      // path when the first operation on the new map is putAll(otherMap). There, prior to
294      // https://github.com/openjdk/jdk/commit/3e393047e12147a81e2899784b943923fc34da8e, a bug
295      // meant that sometimes a too-large threshold is calculated. However, this new threshold is
296      // independent of the initial capacity, except that it won't be lower than the threshold
297      // computed from that capacity. Because the internal table is only allocated on the first
298      // write, we won't see copying because of the new threshold. So it is always OK to use the
299      // calculation here.
300      return (int) Math.ceil(expectedSize / 0.75);
301    }
302    return Integer.MAX_VALUE; // any large value
303  }

471   * a later time.

488      SortedMap<K, ? extends V> sortedLeft = (SortedMap<K, ? extends V>) left;
489      return difference(sortedLeft, right);
490    }
491    return difference(left, right, Equivalence.equals());

504   * @param valueEquivalence the equivalence relationship to use to compare values
505   * @return the difference between the two maps
506   * @since 10.0

508  public static <K extends @Nullable Object, V extends @Nullable Object>
509      MapDifference<K, V> difference(
510          Map<? extends K, ? extends V> left,
511          Map<? extends K, ? extends V> right,
512          Equivalence<? super @NonNull V> valueEquivalence) {
513    Preconditions.checkNotNull(valueEquivalence);
514
515    Map<K, V> onlyOnLeft = newLinkedHashMap();
516    Map<K, V> onlyOnRight = new LinkedHashMap<>(right); // will whittle it down
517    Map<K, V> onBoth = newLinkedHashMap();
518    Map<K, MapDifference.ValueDifference<V>> differences = newLinkedHashMap();
519    doDifference(left, right, valueEquivalence, onlyOnLeft, onlyOnRight, onBoth, differences);
520    return new MapDifferenceImpl<>(onlyOnLeft, onlyOnRight, onBoth, differences);
521  }

523  /**
524   * Computes the difference between two sorted maps, using the comparator of the left map, or
525   * {@code Ordering.natural()} if the left map uses the natural ordering of its elements. This
526   * difference is an immutable snapshot of the state of the maps at the time this method is called.
527   * It will never change, even if the maps change at a later time.
528   *
529   * <p>Since this method uses {@code TreeMap} instances internally, the keys of the right map must
530   * all compare as distinct according to the comparator of the left map.
531   *
532   * <p><b>Note:</b>If you only need to know whether two sorted maps have the same mappings, call
533   * {@code left.equals(right)} instead of this method.
534   *
535   * @param left the map to treat as the "left" map for purposes of comparison
536   * @param right the map to treat as the "right" map for purposes of comparison
537   * @return the difference between the two maps
538   * @since 11.0
539   */
540  public static <K extends @Nullable Object, V extends @Nullable Object>
541      SortedMapDifference<K, V> difference(
542          SortedMap<K, ? extends V> left, Map<? extends K, ? extends V> right) {
543    checkNotNull(left);
544    checkNotNull(right);
545    Comparator<? super K> comparator = orNaturalOrder(left.comparator());
546    SortedMap<K, V> onlyOnLeft = Maps.newTreeMap(comparator);
547    SortedMap<K, V> onlyOnRight = Maps.newTreeMap(comparator);
548    onlyOnRight.putAll(right); // will whittle it down
549    SortedMap<K, V> onBoth = Maps.newTreeMap(comparator);
550    SortedMap<K, MapDifference.ValueDifference<V>> differences = Maps.newTreeMap(comparator);
551
552    doDifference(left, right, Equivalence.equals(), onlyOnLeft, onlyOnRight, onBoth, differences);
553    return new SortedMapDifferenceImpl<>(onlyOnLeft, onlyOnRight, onBoth, differences);
554  }

556  private static <K extends @Nullable Object, V extends @Nullable Object> void doDifference(
557      Map<? extends K, ? extends V> left,
558      Map<? extends K, ? extends V> right,
559      Equivalence<? super @NonNull V> valueEquivalence,
560      Map<K, V> onlyOnLeft,
561      Map<K, V> onlyOnRight,
562      Map<K, V> onBoth,
563      Map<K, MapDifference.ValueDifference<V>> differences) {
564    for (Entry<? extends K, ? extends V> entry : left.entrySet()) {
565      K leftKey = entry.getKey();
566      V leftValue = entry.getValue();
567      if (right.containsKey(leftKey)) {
568        /*
569         * The cast is safe because onlyOnRight contains all the keys of right.
570         *
571         * TODO(cpovirk): Consider checking onlyOnRight.containsKey instead of right.containsKey.
572         * That could change behavior if the input maps use different equivalence relations (and so
573         * a key that appears once in `right` might appear multiple times in `left`). We don't
574         * guarantee behavior in that case, anyway, and the current behavior is likely undesirable.
575         * So that's either a reason to feel free to change it or a reason to not bother thinking
576         * further about this.
577         */
578        V rightValue = uncheckedCastNullableTToT(onlyOnRight.remove(leftKey));
579        if (valueEquivalence.equivalent(leftValue, rightValue)) {
580          onBoth.put(leftKey, leftValue);
581        } else {
582          differences.put(leftKey, ValueDifferenceImpl.create(leftValue, rightValue));
583        }
584      } else {
585        onlyOnLeft.put(leftKey, leftValue);
586      }
587    }
588  }

2795  private abstract static class AbstractFilteredMap<
2796          K extends @Nullable Object, V extends @Nullable Object>
2797      extends ViewCachingAbstractMap<K, V> {
2798    final Map<K, V> unfiltered;
2799    final Predicate<? super Entry<K, V>> predicate;
2800
2801    AbstractFilteredMap(Map<K, V> unfiltered, Predicate<? super Entry<K, V>> predicate) {
2802      this.unfiltered = unfiltered;
2803      this.predicate = predicate;
2804    }
2805
2806    boolean apply(@CheckForNull Object key, @ParametricNullness V value) {
2807      // This method is called only when the key is in the map (or about to be added to the map),
2808      // implying that key is a K.
2809      @SuppressWarnings({"unchecked", "nullness"})
2810      K k = (K) key;
2811      return predicate.apply(Maps.immutableEntry(k, value));
2812    }
2813
2814    @Override
2815    @CheckForNull
2816    public V put(@ParametricNullness K key, @ParametricNullness V value) {
2817      checkArgument(apply(key, value));
2818      return unfiltered.put(key, value);
2819    }
2820
2821    @Override
2822    public void putAll(Map<? extends K, ? extends V> map) {
2823      for (Entry<? extends K, ? extends V> entry : map.entrySet()) {
2824        checkArgument(apply(entry.getKey(), entry.getValue()));
2825      }
2826      unfiltered.putAll(map);
2827    }
2828
2829    @Override
2830    public boolean containsKey(@CheckForNull Object key) {
2831      return unfiltered.containsKey(key) && apply(key, unfiltered.get(key));
2832    }
2833
2834    @Override

4392   * Returns a view of the portion of {@code map} whose keys are contained by {@code range}.
4393   *
4394   * <p>This method delegates to the appropriate methods of {@link NavigableMap} (namely {@link
4395   * NavigableMap#subMap(Object, boolean, Object, boolean) subMap()}, {@link
4396   * NavigableMap#tailMap(Object, boolean) tailMap()}, and {@link NavigableMap#headMap(Object,
4397   * boolean) headMap()}) to actually construct the view. Consult these methods for a full
4398   * description of the returned view's behavior.
4399   *
4400   * <p><b>Warning:</b> {@code Range}s always represent a range of values using the values' natural
4401   * ordering. {@code NavigableMap} on the other hand can specify a custom ordering via a {@link
4402   * Comparator}, which can violate the natural ordering. Using this method (or in general using
4403   * {@code Range}) with unnaturally-ordered maps can lead to unexpected and undefined behavior.
4404   *
4405   * @since 20.0
4406   */

4408  public static <K extends Comparable<? super K>, V extends @Nullable Object>
4409      NavigableMap<K, V> subMap(NavigableMap<K, V> map, Range<K> range) {

4411        && map.comparator() != Ordering.natural()
4412        && range.hasLowerBound()
4413        && range.hasUpperBound()) {
4414      checkArgument(
4415          map.comparator().compare(range.lowerEndpoint(), range.upperEndpoint()) <= 0,
4416          "map is using a custom comparator which is inconsistent with the natural ordering.");
4417    }
4418    if (range.hasLowerBound() && range.hasUpperBound()) {
4419      return map.subMap(
4420          range.lowerEndpoint(),
4421          range.lowerBoundType() == BoundType.CLOSED,
4422          range.upperEndpoint(),
4423          range.upperBoundType() == BoundType.CLOSED);
4424    } else if (range.hasLowerBound()) {
4425      return map.tailMap(range.lowerEndpoint(), range.lowerBoundType() == BoundType.CLOSED);
4426    } else if (range.hasUpperBound()) {
4427      return map.headMap(range.upperEndpoint(), range.upperBoundType() == BoundType.CLOSED);
4428    }
