045/**
046 * An implementation of {@code RangeMap} based on a {@code TreeMap}, supporting all optional
047 * operations.
048 *
049 * <p>Like all {@code RangeMap} implementations, this supports neither null keys nor null values.
050 *
051 * @author Louis Wasserman
052 * @since 14.0
053 */
054@SuppressWarnings("rawtypes") // https://github.com/google/guava/issues/989
055@GwtIncompatible // NavigableMap
056@ElementTypesAreNonnullByDefault
057public final class TreeRangeMap<K extends Comparable, V> implements RangeMap<K, V> {
058
059  private final NavigableMap<Cut<K>, RangeMapEntry<K, V>> entriesByLowerBound;
060
061  public static <K extends Comparable, V> TreeRangeMap<K, V> create() {
062    return new TreeRangeMap<>();
063  }
064
065  @SuppressWarnings("unchecked")
066  public static <K extends Comparable<?>, V> TreeRangeMap<K, V> copyOf(
067      RangeMap<K, ? extends V> rangeMap) {
068    if (rangeMap instanceof TreeRangeMap) {
069      NavigableMap<Cut<K>, RangeMapEntry<K, V>> entriesByLowerBound = Maps.newTreeMap();
070      entriesByLowerBound.putAll(((TreeRangeMap<K, V>) rangeMap).entriesByLowerBound);
071      return new TreeRangeMap<>(entriesByLowerBound);
072    } else {
073      NavigableMap<Cut<K>, RangeMapEntry<K, V>> entriesByLowerBound = Maps.newTreeMap();
074      for (Entry<Range<K>, ? extends V> entry : rangeMap.asMapOfRanges().entrySet()) {
075        entriesByLowerBound.put(
076            entry.getKey().lowerBound(), new RangeMapEntry<K, V>(entry.getKey(), entry.getValue()));
077      }
078      return new TreeRangeMap<>(entriesByLowerBound);
079    }
080  }

090  private static final class RangeMapEntry<K extends Comparable, V>
091      extends AbstractMapEntry<Range<K>, V> {
092    private final Range<K> range;
093    private final V value;
094
095    RangeMapEntry(Cut<K> lowerBound, Cut<K> upperBound, V value) {
096      this(Range.create(lowerBound, upperBound), value);
097    }
098
099    RangeMapEntry(Range<K> range, V value) {
100      this.range = range;
101      this.value = value;
102    }
103
104    @Override
105    public Range<K> getKey() {
106      return range;
107    }
108
109    @Override
110    public V getValue() {
111      return value;
112    }
113
114    public boolean contains(K value) {
115      return range.contains(value);
116    }
117
118    Cut<K> getLowerBound() {
119      return range.lowerBound;
120    }
121
122    Cut<K> getUpperBound() {
123      return range.upperBound;
124    }
125  }
126
127  @Override
128  @CheckForNull
129  public V get(K key) {
130    Entry<Range<K>, V> entry = getEntry(key);
131    return (entry == null) ? null : entry.getValue();
132  }
133
134  @Override
135  @CheckForNull
136  public Entry<Range<K>, V> getEntry(K key) {
137    Entry<Cut<K>, RangeMapEntry<K, V>> mapEntry =
138        entriesByLowerBound.floorEntry(Cut.belowValue(key));
139    if (mapEntry != null && mapEntry.getValue().contains(key)) {
140      return mapEntry.getValue();
141    } else {
142      return null;
143    }
144  }
145
146  @Override
147  public void put(Range<K> range, V value) {
148    if (!range.isEmpty()) {
149      checkNotNull(value);
150      remove(range);
151      entriesByLowerBound.put(range.lowerBound, new RangeMapEntry<K, V>(range, value));
152    }
153  }
154
155  @Override
156  public void putCoalescing(Range<K> range, V value) {
157    // don't short-circuit if the range is empty - it may be between two ranges we can coalesce.
158    if (entriesByLowerBound.isEmpty()) {
159      put(range, value);
160      return;
161    }
162
163    Range<K> coalescedRange = coalescedRange(range, checkNotNull(value));
164    put(coalescedRange, value);
165  }
166
167  /** Computes the coalesced range for the given range+value - does not mutate the map. */
168  private Range<K> coalescedRange(Range<K> range, V value) {
169    Range<K> coalescedRange = range;
170    Entry<Cut<K>, RangeMapEntry<K, V>> lowerEntry =
171        entriesByLowerBound.lowerEntry(range.lowerBound);
172    coalescedRange = coalesce(coalescedRange, value, lowerEntry);
173
174    Entry<Cut<K>, RangeMapEntry<K, V>> higherEntry =
175        entriesByLowerBound.floorEntry(range.upperBound);
176    coalescedRange = coalesce(coalescedRange, value, higherEntry);
177
178    return coalescedRange;
179  }
180
181  /** Returns the range that spans the given range and entry, if the entry can be coalesced. */
182  private static <K extends Comparable, V> Range<K> coalesce(
183      Range<K> range, V value, @CheckForNull Entry<Cut<K>, RangeMapEntry<K, V>> entry) {
184    if (entry != null
185        && entry.getValue().getKey().isConnected(range)
186        && entry.getValue().getValue().equals(value)) {
187      return range.span(entry.getValue().getKey());
188    }
189    return range;
190  }

193  public void putAll(RangeMap<K, ? extends V> rangeMap) {
194    for (Entry<Range<K>, ? extends V> entry : rangeMap.asMapOfRanges().entrySet()) {
195      put(entry.getKey(), entry.getValue());
196    }

204  @Override
205  public Range<K> span() {
206    Entry<Cut<K>, RangeMapEntry<K, V>> firstEntry = entriesByLowerBound.firstEntry();
207    Entry<Cut<K>, RangeMapEntry<K, V>> lastEntry = entriesByLowerBound.lastEntry();
208    // Either both are null or neither is, but we check both to satisfy the nullness checker.
209    if (firstEntry == null || lastEntry == null) {
210      throw new NoSuchElementException();
211    }
212    return Range.create(
213        firstEntry.getValue().getKey().lowerBound, lastEntry.getValue().getKey().upperBound);
214  }

216  private void putRangeMapEntry(Cut<K> lowerBound, Cut<K> upperBound, V value) {
217    entriesByLowerBound.put(lowerBound, new RangeMapEntry<K, V>(lowerBound, upperBound, value));
218  }

220  @Override
221  public void remove(Range<K> rangeToRemove) {
222    if (rangeToRemove.isEmpty()) {
223      return;
224    }
225
226    /*
227     * The comments for this method will use [ ] to indicate the bounds of rangeToRemove and ( ) to
228     * indicate the bounds of ranges in the range map.
229     */
230    Entry<Cut<K>, RangeMapEntry<K, V>> mapEntryBelowToTruncate =
231        entriesByLowerBound.lowerEntry(rangeToRemove.lowerBound);
232    if (mapEntryBelowToTruncate != null) {
233      // we know ( [
234      RangeMapEntry<K, V> rangeMapEntry = mapEntryBelowToTruncate.getValue();
235      if (rangeMapEntry.getUpperBound().compareTo(rangeToRemove.lowerBound) > 0) {
236        // we know ( [ )
237        if (rangeMapEntry.getUpperBound().compareTo(rangeToRemove.upperBound) > 0) {
238          // we know ( [ ] ), so insert the range ] ) back into the map --
239          // it's being split apart
240          putRangeMapEntry(
241              rangeToRemove.upperBound,
242              rangeMapEntry.getUpperBound(),
243              mapEntryBelowToTruncate.getValue().getValue());
244        }
245        // overwrite mapEntryToTruncateBelow with a truncated range
246        putRangeMapEntry(
247            rangeMapEntry.getLowerBound(),
248            rangeToRemove.lowerBound,
249            mapEntryBelowToTruncate.getValue().getValue());
250      }
251    }
252
253    Entry<Cut<K>, RangeMapEntry<K, V>> mapEntryAboveToTruncate =
254        entriesByLowerBound.lowerEntry(rangeToRemove.upperBound);
255    if (mapEntryAboveToTruncate != null) {
256      // we know ( ]
257      RangeMapEntry<K, V> rangeMapEntry = mapEntryAboveToTruncate.getValue();
258      if (rangeMapEntry.getUpperBound().compareTo(rangeToRemove.upperBound) > 0) {
259        // we know ( ] ), and since we dealt with truncating below already,
260        // we know [ ( ] )
261        putRangeMapEntry(
262            rangeToRemove.upperBound,
263            rangeMapEntry.getUpperBound(),
264            mapEntryAboveToTruncate.getValue().getValue());
265      }
266    }
267    entriesByLowerBound.subMap(rangeToRemove.lowerBound, rangeToRemove.upperBound).clear();
268  }

270  private void split(Cut<K> cut) {
271    /*
272     * The comments for this method will use | to indicate the cut point and ( ) to indicate the
273     * bounds of ranges in the range map.
274     */
275    Entry<Cut<K>, RangeMapEntry<K, V>> mapEntryToSplit = entriesByLowerBound.lowerEntry(cut);
276    if (mapEntryToSplit == null) {
277      return;
278    }
279    // we know ( |
280    RangeMapEntry<K, V> rangeMapEntry = mapEntryToSplit.getValue();
281    if (rangeMapEntry.getUpperBound().compareTo(cut) <= 0) {
282      return;
283    }
284    // we know ( | )
285    putRangeMapEntry(rangeMapEntry.getLowerBound(), cut, rangeMapEntry.getValue());
286    putRangeMapEntry(cut, rangeMapEntry.getUpperBound(), rangeMapEntry.getValue());
287  }

292  @Override
293  public void merge(
294      Range<K> range,
295      @CheckForNull V value,
296      BiFunction<? super V, ? super @Nullable V, ? extends @Nullable V> remappingFunction) {
297    checkNotNull(range);
298    checkNotNull(remappingFunction);
299
300    if (range.isEmpty()) {
301      return;
302    }
303    split(range.lowerBound);
304    split(range.upperBound);
305
306    // Due to the splitting of any entries spanning the range bounds, we know that any entry with a
307    // lower bound in the merge range is entirely contained by the merge range.
308    Set<Entry<Cut<K>, RangeMapEntry<K, V>>> entriesInMergeRange =
309        entriesByLowerBound.subMap(range.lowerBound, range.upperBound).entrySet();
310
311    // Create entries mapping any unmapped ranges in the merge range to the specified value.
312    ImmutableMap.Builder<Cut<K>, RangeMapEntry<K, V>> gaps = ImmutableMap.builder();
313    if (value != null) {
314      final Iterator<Entry<Cut<K>, RangeMapEntry<K, V>>> backingItr =
315          entriesInMergeRange.iterator();
316      Cut<K> lowerBound = range.lowerBound;
317      while (backingItr.hasNext()) {
318        RangeMapEntry<K, V> entry = backingItr.next().getValue();
319        Cut<K> upperBound = entry.getLowerBound();
320        if (!lowerBound.equals(upperBound)) {
321          gaps.put(lowerBound, new RangeMapEntry<K, V>(lowerBound, upperBound, value));
322        }
323        lowerBound = entry.getUpperBound();
324      }
325      if (!lowerBound.equals(range.upperBound)) {
326        gaps.put(lowerBound, new RangeMapEntry<K, V>(lowerBound, range.upperBound, value));
327      }
328    }
329
330    // Remap all existing entries in the merge range.
331    final Iterator<Entry<Cut<K>, RangeMapEntry<K, V>>> backingItr = entriesInMergeRange.iterator();
332    while (backingItr.hasNext()) {
333      Entry<Cut<K>, RangeMapEntry<K, V>> entry = backingItr.next();
334      V newValue = remappingFunction.apply(entry.getValue().getValue(), value);
335      if (newValue == null) {
336        backingItr.remove();
337      } else {
338        entry.setValue(
339            new RangeMapEntry<K, V>(
340                entry.getValue().getLowerBound(), entry.getValue().getUpperBound(), newValue));
341      }
342    }
343
344    entriesByLowerBound.putAll(gaps.build());
345  }

489  private class SubRangeMap implements RangeMap<K, V> {
490
491    private final Range<K> subRange;
492
493    SubRangeMap(Range<K> subRange) {
494      this.subRange = subRange;
495    }
496
497    @Override
498    @CheckForNull
499    public V get(K key) {
500      return subRange.contains(key) ? TreeRangeMap.this.get(key) : null;
501    }
502
503    @Override
504    @CheckForNull
505    public Entry<Range<K>, V> getEntry(K key) {
506      if (subRange.contains(key)) {
507        Entry<Range<K>, V> entry = TreeRangeMap.this.getEntry(key);
508        if (entry != null) {
509          return immutableEntry(entry.getKey().intersection(subRange), entry.getValue());
510        }
511      }
512      return null;
513    }
514
515    @Override
516    public Range<K> span() {
517      Cut<K> lowerBound;
518      Entry<Cut<K>, RangeMapEntry<K, V>> lowerEntry =
519          entriesByLowerBound.floorEntry(subRange.lowerBound);
520      if (lowerEntry != null
521          && lowerEntry.getValue().getUpperBound().compareTo(subRange.lowerBound) > 0) {
522        lowerBound = subRange.lowerBound;
523      } else {
524        lowerBound = entriesByLowerBound.ceilingKey(subRange.lowerBound);
525        if (lowerBound == null || lowerBound.compareTo(subRange.upperBound) >= 0) {
526          throw new NoSuchElementException();
527        }
528      }
529
530      Cut<K> upperBound;
531      Entry<Cut<K>, RangeMapEntry<K, V>> upperEntry =
532          entriesByLowerBound.lowerEntry(subRange.upperBound);
533      if (upperEntry == null) {
534        throw new NoSuchElementException();
535      } else if (upperEntry.getValue().getUpperBound().compareTo(subRange.upperBound) >= 0) {
536        upperBound = subRange.upperBound;
537      } else {
538        upperBound = upperEntry.getValue().getUpperBound();
539      }
540      return Range.create(lowerBound, upperBound);
541    }
542
543    @Override
544    public void put(Range<K> range, V value) {
545      checkArgument(
546          subRange.encloses(range), "Cannot put range %s into a subRangeMap(%s)", range, subRange);
547      TreeRangeMap.this.put(range, value);
548    }
549
550    @Override
551    public void putCoalescing(Range<K> range, V value) {
552      if (entriesByLowerBound.isEmpty() || !subRange.encloses(range)) {
553        put(range, value);
554        return;
555      }
556
557      Range<K> coalescedRange = coalescedRange(range, checkNotNull(value));
558      // only coalesce ranges within the subRange
559      put(coalescedRange.intersection(subRange), value);
560    }
561
562    @Override
563    public void putAll(RangeMap<K, ? extends V> rangeMap) {
564      if (rangeMap.asMapOfRanges().isEmpty()) {
565        return;
566      }
567      Range<K> span = rangeMap.span();
568      checkArgument(
569          subRange.encloses(span),
570          "Cannot putAll rangeMap with span %s into a subRangeMap(%s)",
571          span,
572          subRange);
573      TreeRangeMap.this.putAll(rangeMap);
574    }
575
576    @Override
577    public void clear() {
578      TreeRangeMap.this.remove(subRange);
579    }
580
581    @Override
582    public void remove(Range<K> range) {
583      if (range.isConnected(subRange)) {
584        TreeRangeMap.this.remove(range.intersection(subRange));
585      }
586    }
587
588    @Override
589    public void merge(
590        Range<K> range,
591        @CheckForNull V value,
592        BiFunction<? super V, ? super @Nullable V, ? extends @Nullable V> remappingFunction) {
593      checkArgument(
594          subRange.encloses(range),
595          "Cannot merge range %s into a subRangeMap(%s)",
596          range,
597          subRange);
598      TreeRangeMap.this.merge(range, value, remappingFunction);
599    }
600
601    @Override
602    public RangeMap<K, V> subRangeMap(Range<K> range) {
603      if (!range.isConnected(subRange)) {
604        return emptySubRangeMap();
605      } else {
606        return TreeRangeMap.this.subRangeMap(range.intersection(subRange));
607      }
608    }
609
610    @Override
611    public Map<Range<K>, V> asMapOfRanges() {
612      return new SubRangeMapAsMap();
613    }
614
615    @Override
616    public Map<Range<K>, V> asDescendingMapOfRanges() {
617      return new SubRangeMapAsMap() {
618
619        @Override
620        Iterator<Entry<Range<K>, V>> entryIterator() {
621          if (subRange.isEmpty()) {
622            return Iterators.emptyIterator();
623          }
624          final Iterator<RangeMapEntry<K, V>> backingItr =
625              entriesByLowerBound
626                  .headMap(subRange.upperBound, false)
627                  .descendingMap()
628                  .values()
629                  .iterator();
630          return new AbstractIterator<Entry<Range<K>, V>>() {
631
632            @Override
633            @CheckForNull
634            protected Entry<Range<K>, V> computeNext() {
635              if (backingItr.hasNext()) {
636                RangeMapEntry<K, V> entry = backingItr.next();
637                if (entry.getUpperBound().compareTo(subRange.lowerBound) <= 0) {
638                  return endOfData();
639                }
640                return immutableEntry(entry.getKey().intersection(subRange), entry.getValue());
641              }
642              return endOfData();
643            }
644          };
645        }
646      };
647    }

649    @Override
650    public boolean equals(@CheckForNull Object o) {
651      if (o instanceof RangeMap) {
652        RangeMap<?, ?> rangeMap = (RangeMap<?, ?>) o;
653        return asMapOfRanges().equals(rangeMap.asMapOfRanges());
654      }
655      return false;
656    }

664    public String toString() {
665      return asMapOfRanges().toString();
666    }

668    class SubRangeMapAsMap extends AbstractMap<Range<K>, V> {
669
670      @Override
671      public boolean containsKey(@CheckForNull Object key) {
672        return get(key) != null;
673      }
674
675      @Override
676      @CheckForNull
677      public V get(@CheckForNull Object key) {
678        try {
679          if (key instanceof Range) {
680            @SuppressWarnings("unchecked") // we catch ClassCastExceptions
681            Range<K> r = (Range<K>) key;
682            if (!subRange.encloses(r) || r.isEmpty()) {
683              return null;
684            }
685            RangeMapEntry<K, V> candidate = null;
686            if (r.lowerBound.compareTo(subRange.lowerBound) == 0) {
687              // r could be truncated on the left
688              Entry<Cut<K>, RangeMapEntry<K, V>> entry =
689                  entriesByLowerBound.floorEntry(r.lowerBound);
690              if (entry != null) {
691                candidate = entry.getValue();
692              }
693            } else {
694              candidate = entriesByLowerBound.get(r.lowerBound);
695            }
696
697            if (candidate != null
698                && candidate.getKey().isConnected(subRange)
699                && candidate.getKey().intersection(subRange).equals(r)) {
700              return candidate.getValue();
701            }
702          }
703        } catch (ClassCastException e) {
704          return null;
705        }
706        return null;

709      @Override

711      public V remove(@CheckForNull Object key) {
712        V value = get(key);
713        if (value != null) {
714          // it's definitely in the map, so the cast and requireNonNull are safe
715          @SuppressWarnings("unchecked")
716          Range<K> range = (Range<K>) requireNonNull(key);
717          TreeRangeMap.this.remove(range);
718          return value;
719        }
720        return null;
721      }

728      private boolean removeEntryIf(Predicate<? super Entry<Range<K>, V>> predicate) {
729        List<Range<K>> toRemove = Lists.newArrayList();
730        for (Entry<Range<K>, V> entry : entrySet()) {
731          if (predicate.apply(entry)) {
732            toRemove.add(entry.getKey());
733          }
734        }
735        for (Range<K> range : toRemove) {
736          TreeRangeMap.this.remove(range);
737        }
738        return !toRemove.isEmpty();
739      }

786      Iterator<Entry<Range<K>, V>> entryIterator() {
787        if (subRange.isEmpty()) {
788          return Iterators.emptyIterator();
789        }
790        Cut<K> cutToStart =
791            MoreObjects.firstNonNull(
792                entriesByLowerBound.floorKey(subRange.lowerBound), subRange.lowerBound);
793        final Iterator<RangeMapEntry<K, V>> backingItr =
794            entriesByLowerBound.tailMap(cutToStart, true).values().iterator();
795        return new AbstractIterator<Entry<Range<K>, V>>() {
796
797          @Override
798          @CheckForNull
799          protected Entry<Range<K>, V> computeNext() {
800            while (backingItr.hasNext()) {
801              RangeMapEntry<K, V> entry = backingItr.next();
802              if (entry.getLowerBound().compareTo(subRange.upperBound) >= 0) {
803                return endOfData();
804              } else if (entry.getUpperBound().compareTo(subRange.lowerBound) > 0) {
805                // this might not be true e.g. at the start of the iteration
806                return immutableEntry(entry.getKey().intersection(subRange), entry.getValue());
807              }
808            }
809            return endOfData();
810          }
811        };
812      }
