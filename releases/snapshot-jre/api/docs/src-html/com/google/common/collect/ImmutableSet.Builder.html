159  /**
160   * Returns an immutable set containing each of {@code elements}, minus duplicates, in the order
161   * each appears first in the source collection.
162   *
163   * <p><b>Performance note:</b> This method will sometimes recognize that the actual copy operation
164   * is unnecessary; for example, {@code copyOf(copyOf(anArrayList))} will copy the data only once.
165   * This reduces the expense of habitually making defensive copies at API boundaries. However, the
166   * precise conditions for skipping the copy operation are undefined.
167   *
168   * @throws NullPointerException if any of {@code elements} is null
169   * @since 7.0 (source-compatible since 2.0)
170   */
171  public static <E> ImmutableSet<E> copyOf(Collection<? extends E> elements) {
172    /*
173     * TODO(lowasser): consider checking for ImmutableAsList here
174     * TODO(lowasser): consider checking for Multiset here
175     */
176    // Don't refer to ImmutableSortedSet by name so it won't pull in all that code
177    if (elements instanceof ImmutableSet && !(elements instanceof SortedSet)) {
178      @SuppressWarnings("unchecked") // all supported methods are covariant
179      ImmutableSet<E> set = (ImmutableSet<E>) elements;
180      if (!set.isPartialView()) {
181        return set;
182      }
183    } else if (elements instanceof EnumSet) {
184      return copyOfEnumSet((EnumSet) elements);
185    }
186
187    int size = elements.size();
188    if (size == 0) {
189      // We avoid allocating anything.
190      return of();
191    }
192    // Collection<E>.toArray() is required to contain only E instances, and all we do is read them.
193    E[] array = (E[]) elements.toArray();
194    /*
195     * For a Set, we guess that it contains no duplicates. That's just a guess for purpose of
196     * sizing; if the Set uses different equality semantics, it might contain duplicates according
197     * to equals(), and we will deduplicate those properly, albeit at some cost in allocations.
198     */
199    int expectedSize =
200        elements instanceof Set ? array.length : estimatedSizeForUnknownDuplication(array.length);
201    return fromArrayWithExpectedSize(array, expectedSize);
202  }

204  /**
205   * Returns an immutable set containing each of {@code elements}, minus duplicates, in the order
206   * each appears first in the source iterable. This method iterates over {@code elements} only
207   * once.
208   *
209   * <p><b>Performance note:</b> This method will sometimes recognize that the actual copy operation
210   * is unnecessary; for example, {@code copyOf(copyOf(anArrayList))} should copy the data only
211   * once. This reduces the expense of habitually making defensive copies at API boundaries.
212   * However, the precise conditions for skipping the copy operation are undefined.
213   *
214   * @throws NullPointerException if any of {@code elements} is null
215   */
216  public static <E> ImmutableSet<E> copyOf(Iterable<? extends E> elements) {
217    return (elements instanceof Collection)
218        ? copyOf((Collection<? extends E>) elements)
219        : copyOf(elements.iterator());

222  /**
223   * Returns an immutable set containing each of {@code elements}, minus duplicates, in the order
224   * each appears first in the source iterator.
225   *
226   * @throws NullPointerException if any of {@code elements} is null
227   */
228  public static <E> ImmutableSet<E> copyOf(Iterator<? extends E> elements) {
229    // We special-case for 0 or 1 elements, but anything further is madness.
230    if (!elements.hasNext()) {
231      return of();
232    }
233    E first = elements.next();
234    if (!elements.hasNext()) {
235      return of(first);
236    } else {
237      return new ImmutableSet.Builder<E>().add(first).addAll(elements).build();
238    }
239  }

242   * Returns an immutable set containing each of {@code elements}, minus duplicates, in the order
243   * each appears first in the source array.
244   *
245   * @throws NullPointerException if any of {@code elements} is null
246   * @since 3.0
247   */
248  public static <E> ImmutableSet<E> copyOf(E[] elements) {
249    return fromArrayWithExpectedSize(elements, estimatedSizeForUnknownDuplication(elements.length));
250  }

252  private static <E> ImmutableSet<E> fromArrayWithExpectedSize(E[] elements, int expectedSize) {
253    switch (elements.length) {
254      case 0:
255        return of();
256      case 1:
257        return of(elements[0]);
258      default:
259        SetBuilderImpl<E> builder = new RegularSetBuilderImpl<E>(expectedSize);
260        for (int i = 0; i < elements.length; i++) {
261          builder = builder.add(elements[i]);
262        }
263        return builder.review().build();
264    }
265  }

280  public boolean equals(@CheckForNull Object object) {
281    if (object == this) {
282      return true;
283    }
284    if (object instanceof ImmutableSet
285        && isHashCodeFast()
286        && ((ImmutableSet<?>) object).isHashCodeFast()
287        && hashCode() != object.hashCode()) {
288      return false;
289    }
290    return Sets.equalsImpl(this, object);

450  /**
451   * A builder for creating {@code ImmutableSet} instances. Example:
452   *
453   * <pre>{@code
454   * static final ImmutableSet<Color> GOOGLE_COLORS =
455   *     ImmutableSet.<Color>builder()
456   *         .addAll(WEBSAFE_COLORS)
457   *         .add(new Color(0, 191, 255))
458   *         .build();
459   * }</pre>
460   *
461   * <p>Elements appear in the resulting set in the same order they were first added to the builder.
462   *
463   * <p>Building does not change the state of the builder, so it is still possible to add more
464   * elements and to build again.
465   *
466   * @since 2.0
467   */
468  public static class Builder<E> extends ImmutableCollection.Builder<E> {
469    /*
470     * `impl` is null only for instances of the subclass, ImmutableSortedSet.Builder. That subclass
471     * overrides all the methods that access it here. Thus, all the methods here can safely assume
472     * that this field is non-null.
473     */
474    @CheckForNull private SetBuilderImpl<E> impl;
475    boolean forceCopy;
476
477    public Builder() {
478      this(0);
479    }
480
481    Builder(int capacity) {
482      if (capacity > 0) {
483        impl = new RegularSetBuilderImpl<E>(capacity);
484      } else {
485        impl = EmptySetBuilderImpl.instance();
486      }
487    }
488
489    Builder(@SuppressWarnings("unused") boolean subclass) {
490      this.impl = null; // unused
491    }
492
493    @VisibleForTesting
494    void forceJdk() {
495      requireNonNull(impl); // see the comment on the field
496      this.impl = new JdkBackedSetBuilderImpl<E>(impl);
497    }
498
499    final void copyIfNecessary() {
500      if (forceCopy) {
501        copy();
502        forceCopy = false;
503      }
504    }
505
506    void copy() {
507      requireNonNull(impl); // see the comment on the field
508      impl = impl.copy();
509    }
510
511    @Override
512    @CanIgnoreReturnValue
513    public Builder<E> add(E element) {
514      requireNonNull(impl); // see the comment on the field
515      checkNotNull(element);
516      copyIfNecessary();
517      impl = impl.add(element);
518      return this;
519    }
520
521    @Override
522    @CanIgnoreReturnValue
523    public Builder<E> add(E... elements) {
524      super.add(elements);
525      return this;
526    }
527
528    /**
529     * Adds each element of {@code elements} to the {@code ImmutableSet}, ignoring duplicate
530     * elements (only the first duplicate element is added).
531     *
532     * @param elements the elements to add
533     * @return this {@code Builder} object
534     * @throws NullPointerException if {@code elements} is null or contains a null element
535     */
536    @Override
537    @CanIgnoreReturnValue
538    public Builder<E> addAll(Iterable<? extends E> elements) {
539      super.addAll(elements);
540      return this;
541    }
542
543    @Override
544    @CanIgnoreReturnValue
545    public Builder<E> addAll(Iterator<? extends E> elements) {
546      super.addAll(elements);
547      return this;
548    }
549
550    @CanIgnoreReturnValue
551    Builder<E> combine(Builder<E> other) {
552      requireNonNull(impl);
553      requireNonNull(other.impl);
554      /*
555       * For discussion of requireNonNull, see the comment on the field.
556       *
557       * (And I don't believe there's any situation in which we call x.combine(y) when x is a plain
558       * ImmutableSet.Builder but y is an ImmutableSortedSet.Builder (or vice versa). Certainly
559       * ImmutableSortedSet.Builder.combine() is written as if its argument will never be a plain
560       * ImmutableSet.Builder: It casts immediately to ImmutableSortedSet.Builder.)
561       */
562      copyIfNecessary();
563      this.impl = this.impl.combine(other.impl);
564      return this;
565    }
566
567    @Override
568    public ImmutableSet<E> build() {
569      requireNonNull(impl); // see the comment on the field
570      forceCopy = true;
571      impl = impl.review();
572      return impl.build();
573    }
574  }

576  /** Swappable internal implementation of an ImmutableSet.Builder. */
577  private abstract static class SetBuilderImpl<E> {
578    // The first `distinct` elements are non-null.
579    // Since we can never access null elements, we don't mark this nullable.
580    E[] dedupedElements;
581    int distinct;
582
583    @SuppressWarnings("unchecked")
584    SetBuilderImpl(int expectedCapacity) {
585      this.dedupedElements = (E[]) new Object[expectedCapacity];
586      this.distinct = 0;
587    }
588
589    /** Initializes this SetBuilderImpl with a copy of the deduped elements array from toCopy. */
590    SetBuilderImpl(SetBuilderImpl<E> toCopy) {
591      this.dedupedElements = Arrays.copyOf(toCopy.dedupedElements, toCopy.dedupedElements.length);
592      this.distinct = toCopy.distinct;
593    }
594
595    /**
596     * Resizes internal data structures if necessary to store the specified number of distinct
597     * elements.
598     */
599    private void ensureCapacity(int minCapacity) {
600      if (minCapacity > dedupedElements.length) {
601        int newCapacity =
602            ImmutableCollection.Builder.expandedCapacity(dedupedElements.length, minCapacity);
603        dedupedElements = Arrays.copyOf(dedupedElements, newCapacity);
604      }
605    }
606
607    /** Adds e to the insertion-order array of deduplicated elements. Calls ensureCapacity. */
608    final void addDedupedElement(E e) {
609      ensureCapacity(distinct + 1);
610      dedupedElements[distinct++] = e;
611    }
612
613    /**
614     * Adds e to this SetBuilderImpl, returning the updated result. Only use the returned
615     * SetBuilderImpl, since we may switch implementations if e.g. hash flooding is detected.
616     */
617    abstract SetBuilderImpl<E> add(E e);
618
619    /** Adds all the elements from the specified SetBuilderImpl to this SetBuilderImpl. */
620    final SetBuilderImpl<E> combine(SetBuilderImpl<E> other) {
621      SetBuilderImpl<E> result = this;
622      for (int i = 0; i < other.distinct; i++) {
623        /*
624         * requireNonNull is safe because we ensure that the first `distinct` elements have been
625         * populated.
626         */
627        result = result.add(requireNonNull(other.dedupedElements[i]));
628      }
629      return result;
630    }
631
632    /**
633     * Creates a new copy of this SetBuilderImpl. Modifications to that SetBuilderImpl will not
634     * affect this SetBuilderImpl or sets constructed from this SetBuilderImpl via build().
635     */
636    abstract SetBuilderImpl<E> copy();
637
638    /**
639     * Call this before build(). Does a final check on the internal data structures, e.g. shrinking
640     * unnecessarily large structures or detecting previously unnoticed hash flooding.
641     */
642    SetBuilderImpl<E> review() {
643      return this;
644    }
645
646    abstract ImmutableSet<E> build();
647  }

687   * Returns an array size suitable for the backing array of a hash table that uses open addressing
688   * with linear probing in its implementation. The returned size is the smallest power of two that
689   * can hold setSize elements with the desired load factor. Always returns at least setSize + 2.
690   */
691  // TODO(cpovirk): Move to Hashing or something, since it's used elsewhere in the Android version.
692  static int chooseTableSize(int setSize) {
693    setSize = Math.max(setSize, 2);
694    // Correct the size for open addressing to match desired load factor.
695    if (setSize < CUTOFF) {
696      // Round up to the next highest power of 2.
697      int tableSize = Integer.highestOneBit(setSize - 1) << 1;
698      while (tableSize * DESIRED_LOAD_FACTOR < setSize) {
699        tableSize <<= 1;
700      }
701      return tableSize;
702    }
703
704    // The table can't be completely full or we'll get infinite reprobes
705    checkArgument(setSize < MAX_TABLE_SIZE, "collection too large");
706    return MAX_TABLE_SIZE;
707  }

709  /**
710   * Default implementation of the guts of ImmutableSet.Builder, creating an open-addressed hash
711   * table and deduplicating elements as they come, so it only allocates O(max(distinct,
712   * expectedCapacity)) rather than O(calls to add).
713   *
714   * <p>This implementation attempts to detect hash flooding, and if it's identified, falls back to
715   * JdkBackedSetBuilderImpl.
716   */
717  private static final class RegularSetBuilderImpl<E> extends SetBuilderImpl<E> {
718    // null until at least two elements are present
719    @CheckForNull private @Nullable Object[] hashTable;
720    private int maxRunBeforeFallback;
721    private int expandTableThreshold;
722    private int hashCode;
723
724    RegularSetBuilderImpl(int expectedCapacity) {
725      super(expectedCapacity);
726      this.hashTable = null;
727      this.maxRunBeforeFallback = 0;
728      this.expandTableThreshold = 0;
729    }
730
731    RegularSetBuilderImpl(RegularSetBuilderImpl<E> toCopy) {
732      super(toCopy);
733      this.hashTable = (toCopy.hashTable == null) ? null : toCopy.hashTable.clone();
734      this.maxRunBeforeFallback = toCopy.maxRunBeforeFallback;
735      this.expandTableThreshold = toCopy.expandTableThreshold;
736      this.hashCode = toCopy.hashCode;
737    }
738
739    @Override
740    SetBuilderImpl<E> add(E e) {
741      checkNotNull(e);
742      if (hashTable == null) {
743        if (distinct == 0) {
744          addDedupedElement(e);
745          return this;
746        } else {
747          ensureTableCapacity(dedupedElements.length);
748          E elem = dedupedElements[0];
749          distinct--;
750          return insertInHashTable(elem).add(e);
751        }
752      }
753      return insertInHashTable(e);
754    }
755
756    private SetBuilderImpl<E> insertInHashTable(E e) {
757      requireNonNull(hashTable);
758      int eHash = e.hashCode();
759      int i0 = Hashing.smear(eHash);
760      int mask = hashTable.length - 1;
761      for (int i = i0; i - i0 < maxRunBeforeFallback; i++) {
762        int index = i & mask;
763        Object tableEntry = hashTable[index];
764        if (tableEntry == null) {
765          addDedupedElement(e);
766          hashTable[index] = e;
767          hashCode += eHash;
768          ensureTableCapacity(distinct); // rebuilds table if necessary
769          return this;
770        } else if (tableEntry.equals(e)) { // not a new element, ignore
771          return this;
772        }
773      }
774      // we fell out of the loop due to a long run; fall back to JDK impl
775      return new JdkBackedSetBuilderImpl<E>(this).add(e);
776    }
777
778    @Override
779    SetBuilderImpl<E> copy() {
780      return new RegularSetBuilderImpl<E>(this);
781    }
782
783    @Override
784    SetBuilderImpl<E> review() {
785      if (hashTable == null) {
786        return this;
787      }
788      int targetTableSize = chooseTableSize(distinct);
789      if (targetTableSize * 2 < hashTable.length) {
790        hashTable = rebuildHashTable(targetTableSize, dedupedElements, distinct);
791        maxRunBeforeFallback = maxRunBeforeFallback(targetTableSize);
792        expandTableThreshold = (int) (DESIRED_LOAD_FACTOR * targetTableSize);
793      }
794      return hashFloodingDetected(hashTable) ? new JdkBackedSetBuilderImpl<E>(this) : this;
795    }
796
797    @Override
798    ImmutableSet<E> build() {
799      switch (distinct) {
800        case 0:
801          return of();
802        case 1:
803          /*
804           * requireNonNull is safe because we ensure that the first `distinct` elements have been
805           * populated.
806           */
807          return of(requireNonNull(dedupedElements[0]));
808        default:
809          /*
810           * The suppression is safe because we ensure that the first `distinct` elements have been
811           * populated.
812           */
813          @SuppressWarnings("nullness")
814          Object[] elements =
815              (distinct == dedupedElements.length)
816                  ? dedupedElements
817                  : Arrays.copyOf(dedupedElements, distinct);
818          return new RegularImmutableSet<E>(
819              elements, hashCode, requireNonNull(hashTable), hashTable.length - 1);
820      }
821    }
822
823    /** Builds a new open-addressed hash table from the first n objects in elements. */
824    static @Nullable Object[] rebuildHashTable(int newTableSize, Object[] elements, int n) {
825      @Nullable Object[] hashTable = new @Nullable Object[newTableSize];
826      int mask = hashTable.length - 1;
827      for (int i = 0; i < n; i++) {
828        // requireNonNull is safe because we ensure that the first n elements have been populated.
829        Object e = requireNonNull(elements[i]);
830        int j0 = Hashing.smear(e.hashCode());
831        for (int j = j0; ; j++) {
832          int index = j & mask;
833          if (hashTable[index] == null) {
834            hashTable[index] = e;
835            break;
836          }
837        }
838      }
839      return hashTable;
840    }
841
842    void ensureTableCapacity(int minCapacity) {
843      int newTableSize;
844      if (hashTable == null) {
845        newTableSize = chooseTableSize(minCapacity);
846        hashTable = new Object[newTableSize];
847      } else if (minCapacity > expandTableThreshold && hashTable.length < MAX_TABLE_SIZE) {
848        newTableSize = hashTable.length * 2;
849        hashTable = rebuildHashTable(newTableSize, dedupedElements, distinct);
850      } else {
851        return;
852      }
853      maxRunBeforeFallback = maxRunBeforeFallback(newTableSize);
854      expandTableThreshold = (int) (DESIRED_LOAD_FACTOR * newTableSize);
855    }
856
857    /**
858     * We attempt to detect deliberate hash flooding attempts. If one is detected, we fall back to a
859     * wrapper around j.u.HashSet, which has built-in flooding protection. MAX_RUN_MULTIPLIER was
860     * determined experimentally to match our desired probability of false positives.
861     */
862    // NB: yes, this is surprisingly high, but that's what the experiments said was necessary
863    // Raising this number slows the worst-case contains behavior, speeds up hashFloodingDetected,
864    // and reduces the false-positive probability.
865    static final int MAX_RUN_MULTIPLIER = 13;
866
867    /**
868     * Checks the whole hash table for poor hash distribution. Takes O(n) in the worst case, O(n /
869     * log n) on average.
870     *
871     * <p>The online hash flooding detecting in RegularSetBuilderImpl.add can detect e.g. many
872     * exactly matching hash codes, which would cause construction to take O(n^2), but can't detect
873     * e.g. hash codes adversarially designed to go into ascending table locations, which keeps
874     * construction O(n) (as desired) but then can have O(n) queries later.
875     *
876     * <p>If this returns false, then no query can take more than O(log n).
877     *
878     * <p>Note that for a RegularImmutableSet with elements with truly random hash codes, contains
879     * operations take expected O(1) time but with high probability take O(log n) for at least some
880     * element. (https://en.wikipedia.org/wiki/Linear_probing#Analysis)
881     *
882     * <p>This method may return {@code true} even on truly random input, but {@code
883     * ImmutableSetTest} tests that the probability of that is low.
884     */
885    static boolean hashFloodingDetected(@Nullable Object[] hashTable) {
886      int maxRunBeforeFallback = maxRunBeforeFallback(hashTable.length);
887      int mask = hashTable.length - 1;
888
889      // Invariant: all elements at indices in [knownRunStart, knownRunEnd) are nonnull.
890      // If knownRunStart == knownRunEnd, this is vacuously true.
891      // When knownRunEnd exceeds hashTable.length, it "wraps", detecting runs around the end
892      // of the table.
893      int knownRunStart = 0;
894      int knownRunEnd = 0;
895
896      outerLoop:
897      while (knownRunStart < hashTable.length) {
898        if (knownRunStart == knownRunEnd && hashTable[knownRunStart] == null) {
899          if (hashTable[(knownRunStart + maxRunBeforeFallback - 1) & mask] == null) {
900            // There are only maxRunBeforeFallback - 1 elements between here and there,
901            // so even if they were all nonnull, we wouldn't detect a hash flood.  Therefore,
902            // we can skip them all.
903            knownRunStart += maxRunBeforeFallback;
904          } else {
905            knownRunStart++; // the only case in which maxRunEnd doesn't increase by mRBF
906            // happens about f * (1-f) for f = DESIRED_LOAD_FACTOR, so around 21% of the time
907          }
908          knownRunEnd = knownRunStart;
909        } else {
910          for (int j = knownRunStart + maxRunBeforeFallback - 1; j >= knownRunEnd; j--) {
911            if (hashTable[j & mask] == null) {
912              knownRunEnd = knownRunStart + maxRunBeforeFallback;
913              knownRunStart = j + 1;
914              continue outerLoop;
915            }
916          }
917          return true;
918        }
919      }
920      return false;
921    }
922
923    /**
924     * If more than this many consecutive positions are filled in a table of the specified size,
925     * report probable hash flooding. ({@link #hashFloodingDetected} may also report hash flooding
926     * if fewer consecutive positions are filled; see that method for details.)
927     */
928    static int maxRunBeforeFallback(int tableSize) {
929      return MAX_RUN_MULTIPLIER * IntMath.log2(tableSize, RoundingMode.UNNECESSARY);
930    }
931  }

933  /**
934   * SetBuilderImpl version that uses a JDK HashSet, which has built in hash flooding protection.
935   */
936  private static final class JdkBackedSetBuilderImpl<E> extends SetBuilderImpl<E> {
937    private final Set<Object> delegate;
938
939    JdkBackedSetBuilderImpl(SetBuilderImpl<E> toCopy) {
940      super(toCopy); // initializes dedupedElements and distinct
941      delegate = Sets.newHashSetWithExpectedSize(distinct);
942      for (int i = 0; i < distinct; i++) {
943        /*
944         * requireNonNull is safe because we ensure that the first `distinct` elements have been
945         * populated.
946         */
947        delegate.add(requireNonNull(dedupedElements[i]));
948      }
949    }
950
951    @Override
952    SetBuilderImpl<E> add(E e) {
953      checkNotNull(e);
954      if (delegate.add(e)) {
955        addDedupedElement(e);
956      }
957      return this;
958    }
959
960    @Override
961    SetBuilderImpl<E> copy() {
962      return new JdkBackedSetBuilderImpl<>(this);
963    }
964
965    @Override
966    ImmutableSet<E> build() {
967      switch (distinct) {
968        case 0:
969          return of();
970        case 1:
971          /*
972           * requireNonNull is safe because we ensure that the first `distinct` elements have been
973           * populated.
974           */
975          return of(requireNonNull(dedupedElements[0]));
976        default:
977          return new JdkBackedImmutableSet<E>(
978              delegate, ImmutableList.asImmutableList(dedupedElements, distinct));
979      }
980    }
981  }

983  private static int estimatedSizeForUnknownDuplication(int inputElementsIncludingAnyDuplicates) {
984    if (inputElementsIncludingAnyDuplicates
985        < ImmutableCollection.Builder.DEFAULT_INITIAL_CAPACITY) {
986      return inputElementsIncludingAnyDuplicates;
987    }
988    // Guess the size is "halfway between" all duplicates and no duplicates, on a log scale.
989    return Math.max(
990        ImmutableCollection.Builder.DEFAULT_INITIAL_CAPACITY,
991        IntMath.sqrt(inputElementsIncludingAnyDuplicates, RoundingMode.CEILING));
992  }
