090  /**
091   * Creates a <i>mutable</i> {@code ArrayList} instance containing the given elements.
092   *
093   * <p><b>Note:</b> essentially the only reason to use this method is when you will need to add or
094   * remove elements later. Otherwise, for non-null elements use {@link ImmutableList#of()} (for
095   * varargs) or {@link ImmutableList#copyOf(Object[])} (for an array) instead. If any elements
096   * might be null, or you need support for {@link List#set(int, Object)}, use {@link
097   * Arrays#asList}.
098   *
099   * <p>Note that even when you do need the ability to add or remove, this method provides only a
100   * tiny bit of syntactic sugar for {@code newArrayList(}{@link Arrays#asList asList}{@code
101   * (...))}, or for creating an empty list then calling {@link Collections#addAll}. This method is
102   * not actually very useful and will likely be deprecated in the future.
103   */
104  @SafeVarargs
105  @GwtCompatible(serializable = true)
106  public static <E extends @Nullable Object> ArrayList<E> newArrayList(E... elements) {
107    checkNotNull(elements); // for GWT
108    // Avoid integer overflow when a large array is passed in
109    int capacity = computeArrayListCapacity(elements.length);
110    ArrayList<E> list = new ArrayList<>(capacity);
111    Collections.addAll(list, elements);
112    return list;
113  }

328  /** @see Lists#asList(Object, Object[]) */
329  private static class OnePlusArrayList<E extends @Nullable Object> extends AbstractList<E>
330      implements Serializable, RandomAccess {
331    @ParametricNullness final E first;
332    final E[] rest;
333
334    OnePlusArrayList(@ParametricNullness E first, E[] rest) {
335      this.first = first;
336      this.rest = checkNotNull(rest);
337    }
338
339    @Override
340    public int size() {
341      return IntMath.saturatedAdd(rest.length, 1);
342    }
343
344    @Override
345    @ParametricNullness
346    public E get(int index) {
347      // check explicitly so the IOOBE will have the right message
348      checkElementIndex(index, size());
349      return (index == 0) ? first : rest[index - 1];
350    }
351
352    @J2ktIncompatible private static final long serialVersionUID = 0;
353  }

355  /** @see Lists#asList(Object, Object, Object[]) */
356  private static class TwoPlusArrayList<E extends @Nullable Object> extends AbstractList<E>
357      implements Serializable, RandomAccess {
358    @ParametricNullness final E first;
359    @ParametricNullness final E second;
360    final E[] rest;
361
362    TwoPlusArrayList(@ParametricNullness E first, @ParametricNullness E second, E[] rest) {
363      this.first = first;
364      this.second = second;
365      this.rest = checkNotNull(rest);
366    }
367
368    @Override
369    public int size() {
370      return IntMath.saturatedAdd(rest.length, 2);
371    }
372
373    @Override
374    @ParametricNullness
375    public E get(int index) {
376      switch (index) {
377        case 0:
378          return first;
379        case 1:
380          return second;
381        default:
382          // check explicitly so the IOOBE will have the right message
383          checkElementIndex(index, size());
384          return rest[index - 2];
385      }
386    }
387
388    @J2ktIncompatible private static final long serialVersionUID = 0;
389  }

391  /**
392   * Returns every possible list that can be formed by choosing one element from each of the given
393   * lists in order; the "n-ary <a href="http://en.wikipedia.org/wiki/Cartesian_product">Cartesian
394   * product</a>" of the lists. For example:
395   *
396   * <pre>{@code
397   * Lists.cartesianProduct(ImmutableList.of(
398   *     ImmutableList.of(1, 2),
399   *     ImmutableList.of("A", "B", "C")))
400   * }</pre>
401   *
402   * <p>returns a list containing six lists in the following order:
403   *
404   * <ul>
405   *   <li>{@code ImmutableList.of(1, "A")}
406   *   <li>{@code ImmutableList.of(1, "B")}
407   *   <li>{@code ImmutableList.of(1, "C")}
408   *   <li>{@code ImmutableList.of(2, "A")}
409   *   <li>{@code ImmutableList.of(2, "B")}
410   *   <li>{@code ImmutableList.of(2, "C")}
411   * </ul>
412   *
413   * <p>The result is guaranteed to be in the "traditional", lexicographical order for Cartesian
414   * products that you would get from nesting for loops:
415   *
416   * <pre>{@code
417   * for (B b0 : lists.get(0)) {
418   *   for (B b1 : lists.get(1)) {
419   *     ...
420   *     ImmutableList<B> tuple = ImmutableList.of(b0, b1, ...);
421   *     // operate on tuple
422   *   }
423   * }
424   * }</pre>
425   *
426   * <p>Note that if any input list is empty, the Cartesian product will also be empty. If no lists
427   * at all are provided (an empty list), the resulting Cartesian product has one element, an empty
428   * list (counter-intuitive, but mathematically consistent).
429   *
430   * <p><i>Performance notes:</i> while the cartesian product of lists of size {@code m, n, p} is a
431   * list of size {@code m x n x p}, its actual memory consumption is much smaller. When the
432   * cartesian product is constructed, the input lists are merely copied. Only as the resulting list
433   * is iterated are the individual lists created, and these are not retained after iteration.
434   *
435   * @param lists the lists to choose elements from, in the order that the elements chosen from
436   *     those lists should appear in the resulting lists
437   * @param <B> any common base class shared by all axes (often just {@link Object})
438   * @return the Cartesian product, as an immutable list containing immutable lists
439   * @throws IllegalArgumentException if the size of the cartesian product would be greater than
440   *     {@link Integer#MAX_VALUE}
441   * @throws NullPointerException if {@code lists}, any one of the {@code lists}, or any element of
442   *     a provided list is null
443   * @since 19.0
444   */
445  public static <B> List<List<B>> cartesianProduct(List<? extends List<? extends B>> lists) {
446    return CartesianList.create(lists);
447  }

508  /**
509   * Returns a list that applies {@code function} to each element of {@code fromList}. The returned
510   * list is a transformed view of {@code fromList}; changes to {@code fromList} will be reflected
511   * in the returned list and vice versa.
512   *
513   * <p>Since functions are not reversible, the transform is one-way and new items cannot be stored
514   * in the returned list. The {@code add}, {@code addAll} and {@code set} methods are unsupported
515   * in the returned list.
516   *
517   * <p>The function is applied lazily, invoked when needed. This is necessary for the returned list
518   * to be a view, but it means that the function will be applied many times for bulk operations
519   * like {@link List#contains} and {@link List#hashCode}. For this to perform well, {@code
520   * function} should be fast. To avoid lazy evaluation when the returned list doesn't need to be a
521   * view, copy the returned list into a new list of your choosing.
522   *
523   * <p>If {@code fromList} implements {@link RandomAccess}, so will the returned list. The returned
524   * list is threadsafe if the supplied list and function are.
525   *
526   * <p>If only a {@code Collection} or {@code Iterable} input is available, use {@link
527   * Collections2#transform} or {@link Iterables#transform}.
528   *
529   * <p><b>Note:</b> serializing the returned list is implemented by serializing {@code fromList},
530   * its contents, and {@code function} -- <i>not</i> by serializing the transformed values. This
531   * can lead to surprising behavior, so serializing the returned list is <b>not recommended</b>.
532   * Instead, copy the list using {@link ImmutableList#copyOf(Collection)} (for example), then
533   * serialize the copy. Other methods similar to this do not implement serialization at all for
534   * this reason.
535   *
536   * <p><b>Java 8+ users:</b> many use cases for this method are better addressed by {@link
537   * java.util.stream.Stream#map}. This method is not being deprecated, but we gently encourage you
538   * to migrate to streams.
539   */
540  public static <F extends @Nullable Object, T extends @Nullable Object> List<T> transform(
541      List<F> fromList, Function<? super F, ? extends T> function) {
542    return (fromList instanceof RandomAccess)
543        ? new TransformingRandomAccessList<>(fromList, function)
544        : new TransformingSequentialList<>(fromList, function);
545  }

547  /**
548   * Implementation of a sequential transforming list.
549   *
550   * @see Lists#transform
551   */
552  private static class TransformingSequentialList<
553          F extends @Nullable Object, T extends @Nullable Object>
554      extends AbstractSequentialList<T> implements Serializable {
555    final List<F> fromList;
556    final Function<? super F, ? extends T> function;
557
558    TransformingSequentialList(List<F> fromList, Function<? super F, ? extends T> function) {
559      this.fromList = checkNotNull(fromList);
560      this.function = checkNotNull(function);
561    }
562
563    /**
564     * The default implementation inherited is based on iteration and removal of each element which
565     * can be overkill. That's why we forward this call directly to the backing list.
566     */
567    @Override
568    protected void removeRange(int fromIndex, int toIndex) {
569      fromList.subList(fromIndex, toIndex).clear();
570    }

675  /**
676   * Returns consecutive {@linkplain List#subList(int, int) sublists} of a list, each of the same
677   * size (the final list may be smaller). For example, partitioning a list containing {@code [a, b,
678   * c, d, e]} with a partition size of 3 yields {@code [[a, b, c], [d, e]]} -- an outer list
679   * containing two inner lists of three and two elements, all in the original order.
680   *
681   * <p>The outer list is unmodifiable, but reflects the latest state of the source list. The inner
682   * lists are sublist views of the original list, produced on demand using {@link List#subList(int,
683   * int)}, and are subject to all the usual caveats about modification as explained in that API.
684   *
685   * @param list the list to return consecutive sublists of
686   * @param size the desired size of each sublist (the last may be smaller)
687   * @return a list of consecutive sublists
688   * @throws IllegalArgumentException if {@code partitionSize} is nonpositive
689   */
690  public static <T extends @Nullable Object> List<List<T>> partition(List<T> list, int size) {
691    checkNotNull(list);
692    checkArgument(size > 0);
693    return (list instanceof RandomAccess)
694        ? new RandomAccessPartition<>(list, size)
695        : new Partition<>(list, size);
696  }

699    final List<T> list;

826   * Returns a reversed view of the specified list. For example, {@code
827   * Lists.reverse(Arrays.asList(1, 2, 3))} returns a list containing {@code 3, 2, 1}. The returned
828   * list is backed by this list, so changes in the returned list are reflected in this list, and
829   * vice-versa. The returned list supports all of the optional list operations supported by this
830   * list.
831   *
832   * <p>The returned list is random-access if the specified list is random access.
833   *
834   * @since 7.0
835   */
836  public static <T extends @Nullable Object> List<T> reverse(List<T> list) {
837    if (list instanceof ImmutableList) {
838      // Avoid nullness warnings.
839      List<?> reversed = ((ImmutableList<?>) list).reverse();
840      @SuppressWarnings("unchecked")
841      List<T> result = (List<T>) reversed;
842      return result;
843    } else if (list instanceof ReverseList) {
844      return ((ReverseList<T>) list).getForwardList();
845    } else if (list instanceof RandomAccess) {
846      return new RandomAccessReverseList<>(list);
847    } else {
848      return new ReverseList<>(list);
849    }
850  }

852  private static class ReverseList<T extends @Nullable Object> extends AbstractList<T> {
853    private final List<T> forwardList;
854
855    ReverseList(List<T> forwardList) {
856      this.forwardList = checkNotNull(forwardList);
857    }
858
859    List<T> getForwardList() {
860      return forwardList;
861    }
862
863    private int reverseIndex(int index) {
864      int size = size();
865      checkElementIndex(index, size);
866      return (size - 1) - index;
867    }
868
869    private int reversePosition(int index) {
870      int size = size();
871      checkPositionIndex(index, size);
872      return size - index;
873    }
874
875    @Override
876    public void add(int index, @ParametricNullness T element) {
877      forwardList.add(reversePosition(index), element);
878    }
879
880    @Override
881    public void clear() {
882      forwardList.clear();
883    }
884
885    @Override
886    @ParametricNullness
887    public T remove(int index) {
888      return forwardList.remove(reverseIndex(index));
889    }
890
891    @Override
892    protected void removeRange(int fromIndex, int toIndex) {
893      subList(fromIndex, toIndex).clear();
894    }
895
896    @Override
897    @ParametricNullness
898    public T set(int index, @ParametricNullness T element) {
899      return forwardList.set(reverseIndex(index), element);
900    }
901
902    @Override
903    @ParametricNullness
904    public T get(int index) {
905      return forwardList.get(reverseIndex(index));
906    }
907
908    @Override
909    public int size() {
910      return forwardList.size();
911    }
912
913    @Override
914    public List<T> subList(int fromIndex, int toIndex) {
915      checkPositionIndexes(fromIndex, toIndex, size());
916      return reverse(forwardList.subList(reversePosition(toIndex), reversePosition(fromIndex)));
917    }
918
919    @Override
920    public Iterator<T> iterator() {
921      return listIterator();
922    }
923
924    @Override
925    public ListIterator<T> listIterator(int index) {
926      int start = reversePosition(index);
927      final ListIterator<T> forwardIterator = forwardList.listIterator(start);
928      return new ListIterator<T>() {
929
930        boolean canRemoveOrSet;
931
932        @Override
933        public void add(@ParametricNullness T e) {
934          forwardIterator.add(e);
935          forwardIterator.previous();
936          canRemoveOrSet = false;
937        }
938
939        @Override
940        public boolean hasNext() {
941          return forwardIterator.hasPrevious();
942        }
943
944        @Override
945        public boolean hasPrevious() {
946          return forwardIterator.hasNext();
947        }
948
949        @Override
950        @ParametricNullness
951        public T next() {
952          if (!hasNext()) {
953            throw new NoSuchElementException();
954          }
955          canRemoveOrSet = true;
956          return forwardIterator.previous();
957        }
958
959        @Override
960        public int nextIndex() {
961          return reversePosition(forwardIterator.nextIndex());
962        }
963
964        @Override
965        @ParametricNullness
966        public T previous() {
967          if (!hasPrevious()) {
968            throw new NoSuchElementException();
969          }
970          canRemoveOrSet = true;
971          return forwardIterator.next();
972        }
973
974        @Override
975        public int previousIndex() {
976          return nextIndex() - 1;
977        }
978
979        @Override
980        public void remove() {
981          checkRemove(canRemoveOrSet);
982          forwardIterator.remove();
983          canRemoveOrSet = false;
984        }
985
986        @Override
987        public void set(@ParametricNullness T e) {
988          checkState(canRemoveOrSet);
989          forwardIterator.set(e);
990        }
991      };
992    }
993  }

1003  static int hashCodeImpl(List<?> list) {
1004    // TODO(lowasser): worth optimizing for RandomAccess?
1005    int hashCode = 1;
1006    for (Object o : list) {
1007      hashCode = 31 * hashCode + (o == null ? 0 : o.hashCode());
1008
1009      hashCode = ~~hashCode;
1010      // needed to deal with GWT integer overflow
1011    }

1015  /** An implementation of {@link List#equals(Object)}. */
1016  static boolean equalsImpl(List<?> thisList, @CheckForNull Object other) {
1017    if (other == checkNotNull(thisList)) {
1018      return true;
1019    }
1020    if (!(other instanceof List)) {
1021      return false;
1022    }
1023    List<?> otherList = (List<?>) other;
1024    int size = thisList.size();
1025    if (size != otherList.size()) {
1026      return false;
1027    }
1028    if (thisList instanceof RandomAccess && otherList instanceof RandomAccess) {
1029      // avoid allocation and use the faster loop
1030      for (int i = 0; i < size; i++) {
1031        if (!Objects.equal(thisList.get(i), otherList.get(i))) {
1032          return false;
1033        }
1034      }
1035      return true;
1036    } else {
1037      return Iterators.elementsEqual(thisList.iterator(), otherList.iterator());
1038    }
1039  }

1053  /** An implementation of {@link List#indexOf(Object)}. */
1054  static int indexOfImpl(List<?> list, @CheckForNull Object element) {
1055    if (list instanceof RandomAccess) {
1056      return indexOfRandomAccess(list, element);
1057    } else {
1058      ListIterator<?> listIterator = list.listIterator();
1059      while (listIterator.hasNext()) {
1060        if (Objects.equal(element, listIterator.next())) {
1061          return listIterator.previousIndex();
1062        }
1063      }
1064      return -1;
1065    }
1066  }

1068  private static int indexOfRandomAccess(List<?> list, @CheckForNull Object element) {
1069    int size = list.size();
1070    if (element == null) {
1071      for (int i = 0; i < size; i++) {
1072        if (list.get(i) == null) {
1073          return i;
1074        }
1075      }
1076    } else {
1077      for (int i = 0; i < size; i++) {
1078        if (element.equals(list.get(i))) {
1079          return i;
1080        }
1081      }
1082    }
1083    return -1;
1084  }

1086  /** An implementation of {@link List#lastIndexOf(Object)}. */
1087  static int lastIndexOfImpl(List<?> list, @CheckForNull Object element) {
1088    if (list instanceof RandomAccess) {
1089      return lastIndexOfRandomAccess(list, element);
1090    } else {
1091      ListIterator<?> listIterator = list.listIterator(list.size());
1092      while (listIterator.hasPrevious()) {
1093        if (Objects.equal(element, listIterator.previous())) {
1094          return listIterator.nextIndex();
1095        }
1096      }
1097      return -1;
1098    }
1099  }

1101  private static int lastIndexOfRandomAccess(List<?> list, @CheckForNull Object element) {
1102    if (element == null) {
1103      for (int i = list.size() - 1; i >= 0; i--) {
1104        if (list.get(i) == null) {
1105          return i;
1106        }
1107      }
1108    } else {
1109      for (int i = list.size() - 1; i >= 0; i--) {
1110        if (element.equals(list.get(i))) {
1111          return i;
1112        }
1113      }
1114    }
1115    return -1;
1116  }
